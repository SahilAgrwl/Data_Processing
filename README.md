#Tech Stacks Used :
#For backend : python, django
#For frontend : react

#Environment setup 
Install below libraries before running app :
#For backend :
pip install django, 
pip install django-cors-headers, 
pip install pandas, 

#For Frontend :
npm install axios

#Running 
To run app, use below commands:
#For backend :
pip manage.py makemigrations, 
pip manage.py migrate ,
pip manage.py runserver,

#For Frontend :
npm start
